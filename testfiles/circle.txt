function circle x, y
	setpos x y

	save ocho 0.0
	save seis 0.0
	save nov 0.0
	
	for[ocho, 8.0, 1.0][
		rightturn 45.0,
		for[seis, 6.0, 1.0][
			for[nov, 90.0, 1.0][
				move 1.0,
				rightturn 2.0
			],
			rightturn 90.0
		]
	]

	setpos x y
	leftturn 182.0

	save ocho 0.0
	save seis 0.0
	save nov 0.0

		for[ocho, 8.0, 1.0][
		rightturn 45.0,
		for[seis, 6.0, 1.0][
			for[nov, 90.0, 1.0][
				move 1.0,
				leftturn 2.0
			],
			rightturn 90.0
		]
	]

end

program main
	setbackgroundtxt 2.0
	setcolor 1.0 0.0 0.0
	save i 0.0-450.0
	save j 0.0

	for[i, 250.0, 50.0][
		setcolor j j j,
		circle(i 0.0),
		save j j+0.10
		]
	home
end
